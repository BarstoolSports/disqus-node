create

Creates a new category.

Signature:

Disqus#category.create(options[, cb])

Usage:

// Node-style
disqus.category.create({
    forum: 'forumId',
    title: 'my new category,
    default: false
}, function (err, result) {...})

// Promise-style
disqus.category.create({
    forum: 'forumId',
    title: 'my new category,
    default: false
})
.then(function (result) {...})
.catch(function (err) {...});
this.create = function (options, cb) { return Promise.resolve({}) .then(function (qs) { options = options || {}; qs.api_key = options.api_key || config.api_key; if ('forum' in options) { qs.forum = options.forum; } if ('title' in options) { qs.title = options.title; } if ('default' in options) { qs.default = options.default; } var opts = { method: 'POST', url: config.url + 'categories/create.json', qs: qs }; logger.debug(JSON.stringify(opts)); return request(opts).then(parse); }).nodeify(cb); };
